81 {
82  Real w_h = 0.04*L;
83  Real s_b = w_h;
84  Real s_t = 0.95*L;
85  Real w_t = 0.01*L;
86  Real y = 0.0;
87  if (x >= 0.0 && x < s_b) {
88  y = sqrt(2.0*w_h*x - x * x);
89  }
90  else
91  {
92  if (x >= s_t && x <= L) {
93  y = w_t * (L - x) / (L - s_t);
94  }
95  else
96  {
97  y = w_h - (w_h - w_t)*(x - s_b) / (s_t - s_b);
98  }
99  }
