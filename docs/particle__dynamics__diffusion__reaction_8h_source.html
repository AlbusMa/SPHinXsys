139  void getDiffusionChangeRate(size_t particle_i, size_t particle_j, Vecd& e_ij, Real surface_area_ij)
140  {
141  for (size_t m = 0; m < species_diffusion_.size(); ++m)
142  {
143  Real diff_coff_ij = species_diffusion_[m]->getInterParticleDiffusionCoff(particle_i, particle_j, e_ij);
144  size_t l = species_diffusion_[m]->gradient_species_index_;
145  Real phi_ij = species_n_[l][particle_i] - species_n_[l][particle_j];
146  diffusion_dt_[m][particle_i] += diff_coff_ij * phi_ij * surface_area_ij;
147  }
148  };
149 
155  virtual void updateSpeciesDiffusion(size_t particle_i, Real dt)
156  {
157  for (size_t m = 0; m < species_diffusion_.size(); ++m)
158  {
159  size_t k = species_diffusion_[m]->diffusion_species_index_;
160  species_n_[k][particle_i] += dt * diffusion_dt_[m][particle_i];
161  }
162  };
163 
164  virtual void Interaction(size_t index_i, Real dt = 0.0) override
165  {
166  DiffusionReactionParticles<BaseParticlesType, BaseMaterialType>* particles = this->particles_;
167  Neighborhood& inner_neighborhood = this->inner_configuration_[index_i];
168 
169  initializeDiffusionChangeRate(index_i);
170  for (size_t n = 0; n != inner_neighborhood.current_size_; ++n)
171  {
172  size_t index_j = inner_neighborhood.j_[n];
173  Real dW_ij_ = inner_neighborhood.dW_ij_[n];
174  Real r_ij_ = inner_neighborhood.r_ij_[n];
175  Vecd& e_ij = inner_neighborhood.e_ij_[n];
176 
177  const Vecd& gradi_ij = particles->getKernelGradient(index_i, index_j, dW_ij_, e_ij);
178  Real area_ij = 2.0 * Vol_[index_j] * dot(gradi_ij, e_ij) /r_ij_;
179  getDiffusionChangeRate(index_i, index_j, e_ij, area_ij);
180  }
181  };
182 
183  virtual void Update(size_t index_i, Real dt = 0.0) override
184  {
185  updateSpeciesDiffusion(index_i, dt);
186  };
187  public:
188  RelaxationOfAllDiffussionSpecies(SPHBodyInnerRelation* body_inner_relation)
189  : InteractionDynamicsWithUpdate(body_inner_relation->sph_body_),
190  DiffusionReactionDataDelegateInner<BodyType, BaseParticlesType, BaseMaterialType>(body_inner_relation),
191  species_n_(this->particles_->species_n_),
192  diffusion_dt_(this->particles_->diffusion_dt_), Vol_(this->particles_->Vol_)
193  {
194  species_diffusion_ = this->material_->SpeciesDiffusion();
195  };
196  virtual ~RelaxationOfAllDiffussionSpecies() {};
197  };

240  class RungeKutta2Stages2ndStage :
241  public RelaxationOfAllDiffussionSpecies<BodyType, BaseParticlesType, BaseMaterialType>
242  {
244  StdVec<BaseDiffusion*> species_diffusion_;
245  StdVec<StdLargeVec<Real>>& species_n_;
