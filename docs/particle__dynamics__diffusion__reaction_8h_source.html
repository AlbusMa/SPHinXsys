108  void getDiffusionChangeRate(size_t particle_i, size_t particle_j, Vecd& e_ij, Real surface_area_ij,
109  DiffusionReactionData& diffusion_reaction_data_i, DiffusionReactionData& diffusion_reaction_data_j)
110  {

114  size_t k = species_diffusion_[m]->diffusion_species_index_;
115  size_t l = species_diffusion_[m]->gradient_species_index_;
116  Real phi_ij = diffusion_reaction_data_i.species_n_[l] - diffusion_reaction_data_j.species_n_[l];
117  diffusion_reaction_data_i.dspecies_dt_[k] += diff_coff_ij * phi_ij * surface_area_ij;

135  virtual void InnerInteraction(size_t index_particle_i, Real dt = 0.0) override
136  {
137  DiffusionReactionParticles<BaseParticlesType, BaseMaterialType>* particles = this->particles_;
138  Neighborhood& inner_neighborhood = this->inner_configuration_[index_particle_i];
139 
140  StdLargeVec<DiffusionReactionData>& diffusion_reaction_data = particles->diffusion_reaction_data_;
141  StdLargeVec<BaseParticleData>& base_particle_data = particles->base_particle_data_;
142  DiffusionReactionData& diffusion_reaction_data_i = diffusion_reaction_data[index_particle_i];
143 
144  initializeDiffusionChangeRate(diffusion_reaction_data_i);
145  CommonRelationList& inner_common_relations = inner_neighborhood.common_relation_list_;
146  for (size_t n = 0; n != inner_neighborhood.current_size_; ++n)
147  {
148  CommonRelation& common_relation = inner_common_relations[n];
149  size_t index_particle_j = common_relation.j_;
150  Vecd& e_ij = common_relation.e_ij_;
151  DiffusionReactionData& diffusion_reaction_data_j = diffusion_reaction_data[index_particle_j];
152 
153  const Vecd& gradi_ij = particles->getKernelGradient(index_particle_i, index_particle_j, common_relation.dW_ij_, e_ij);
154  Real area_ij = 2.0 * base_particle_data[index_particle_j].Vol_ * dot(gradi_ij, e_ij) / common_relation.r_ij_;
155  getDiffusionChangeRate(index_particle_i, index_particle_j, e_ij, area_ij,
156  diffusion_reaction_data_i, diffusion_reaction_data_j);
157  }
